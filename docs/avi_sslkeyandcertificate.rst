Examples
--------

.. code-block:: yaml
        
    - name: Create a SSL Key and Certificate
      vmware.alb.avi_sslkeyandcertificate:
        controller: 192.168.138.18
        username: admin
        password: password
        key: |
            -----BEGIN PRIVATE KEY-----
            ....
            -----END PRIVATE KEY-----
        certificate:
            self_signed: true
            certificate: |
              -----BEGIN CERTIFICATE-----
              ....
              -----END CERTIFICATE-----
        type: SSL_CERTIFICATE_TYPE_VIRTUALSERVICE
        name: MyTestCert
