Examples
--------

.. code-block:: yaml
        
      - name: Create an IP Address Group configuration
        vmware.alb.avi_ipaddrgroup:
          controller: '{{ controller }}'
          username: '{{ username }}'
          password: '{{ password }}'
          name: Client-Source-Block
          prefixes:
          - ip_addr:
              addr: 192.168.138.18
              type: V4
            mask: 8
          - ip_addr:
              addr: 192.168.20.11
              type: V4
            mask: 12
          - ip_addr:
              addr: 192.168.20.12
              type: V4
            mask: 16
